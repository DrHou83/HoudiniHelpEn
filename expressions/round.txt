#type: expression
#category: math

= round =

"""Rounds a number to the nearest integer."""

:usage: `round()`

@examples
    {{{
    #!exp
    round(2.501)=3
    }}}
    
    {{{
    #!exp
    round(-2.501)=-3
    }}}
    
    {{{
    #!exp
    round(0.2)=0
    }}}
    
    {{{
    #!exp
    round(-.2)=0
    }}}
