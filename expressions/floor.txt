#type: expression
#category: math

= floor =

"""Returns the largest integer not greater than a number."""

:usage: `floor(<<number>>)`

@examples
    {{{
    #!exp
    floor(10.2)=10
    }}}
    
    {{{
    #!exp
    floor(-10.2)=-11
    }}}
