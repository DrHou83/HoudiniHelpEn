#type: hscript

= tmgadd =

"""Creates time groups."""

:usage: `tmgadd -t <<time>> <<group_name>> [<<group_name2>> ...]`
:usage: `tmgadd -f <<frame>> <<group_name>> [<<group_name2>> ...]`

@options
    `-t <<time>> `:
        Make <<time>> the first member of the new group(s).
    `-f <<frame>> `:
        Make <<frame>> the first member of the new group(s).
